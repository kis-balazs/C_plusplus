// használat elott a codeblocks include könyvtárába másoljuk a graphics.h és winbgim.h állományokat
// a lib könyvtárba a libbgi.a állományt

#include <graphics.h>
#include <iostream>
using namespace std;
int main() {
  int gdriver = DETECT, gmode;
  initgraph(&gdriver, &gmode, "");
  if (graphresult()) {
   cout<<"Tentativa nereusita ...";
  }
  else {
   cout<<"Totul functioneaza corect, afisez!";
   setcolor(RED);
   moveto(0,0);
   lineto(getmaxx(),getmaxy());
  }
  cout<<endl<<"Apasa o tasta pentru a inchide...";
  getch();
  return 0;
}
